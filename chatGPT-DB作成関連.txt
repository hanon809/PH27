物理名称、データ型、桁数、not nullの順番 で書いてあるんだけど、これをCREATE TABLE文にしてほしいな
論理名称付きでお願い



-- ==========================
-- カテゴリー
-- ==========================
CREATE TABLE category (
    f_category_id   INT(11) NOT NULL AUTO_INCREMENT COMMENT 'カテゴリーID',
    f_category_name VARCHAR(50) NOT NULL COMMENT 'カテゴリー名',
    PRIMARY KEY (f_category_id)
) COMMENT='カテゴリー';

-- ==========================
-- ユーザ情報
-- ==========================
CREATE TABLE user (
    f_user_id           INT(11) NOT NULL AUTO_INCREMENT COMMENT 'ユーザID',
    f_user_name         VARCHAR(100) NOT NULL COMMENT 'ユーザ名',
    f_user_mail         VARCHAR(255) NOT NULL UNIQUE COMMENT 'メールアドレス',
    f_phone_number      VARCHAR(20) NULL COMMENT '電話番号',
    f_payment           VARCHAR(50) NULL COMMENT '支払方法',
    f_password          VARCHAR(255) NOT NULL COMMENT 'パスワード',
    f_purchase_history  VARCHAR(255) NULL COMMENT '購入履歴',
    f_address           VARCHAR(255) NULL COMMENT '住所',
    f_postal_code       VARCHAR(10) NULL COMMENT '郵便番号',
    f_registration_time DATETIME NOT NULL COMMENT '登録日時',
    f_delete            BOOLEAN NOT NULL DEFAULT 0 COMMENT '削除フラグ',
    f_update            DATETIME NULL COMMENT '更新日時',
    PRIMARY KEY (f_user_id)
) COMMENT='ユーザ情報';

-- ==========================
-- 管理者
-- ==========================
CREATE TABLE admin (
    f_admin_id INT(11) NOT NULL AUTO_INCREMENT COMMENT '管理者ID',
    f_name     VARCHAR(100) NOT NULL COMMENT '名前',
    f_mail     VARCHAR(255) NOT NULL UNIQUE COMMENT 'メール',
    f_password VARCHAR(255) NOT NULL COMMENT 'パスワード',
    PRIMARY KEY (f_admin_id)
) COMMENT='管理者';

-- ==========================
-- 商品
-- ==========================
CREATE TABLE product (
    f_product_id    INT(11) NOT NULL AUTO_INCREMENT COMMENT '商品ID',
    f_category_id   INT(11) NOT NULL COMMENT 'カテゴリーID',
    f_product_name  VARCHAR(100) NOT NULL COMMENT '商品名',
    f_product_photo VARCHAR(255) NULL COMMENT '商品画像',
    f_product_price DECIMAL(10,2) NOT NULL COMMENT '商品価格',
    f_writer_name   VARCHAR(100) NULL COMMENT '作家名',
    f_product_size  VARCHAR(50) NULL COMMENT '大きさ',
    f_release_date  DATE NULL COMMENT '発売日',
    f_publisher     VARCHAR(100) NULL COMMENT '出版社',
    PRIMARY KEY (f_product_id),
    FOREIGN KEY (f_category_id) REFERENCES category(f_category_id)
) COMMENT='商品';

-- ==========================
-- 注文情報
-- ==========================
CREATE TABLE order_info (
    f_order_id     INT(11) NOT NULL AUTO_INCREMENT COMMENT '注文ID',
    f_user_id      INT(11) NOT NULL COMMENT 'ユーザID',
    f_total_amount DECIMAL(10,2) NOT NULL COMMENT '合計金額',
    f_order_status VARCHAR(50) NOT NULL COMMENT '注文ステータス',
    f_order_time   DATETIME NOT NULL COMMENT '注文日時',
    PRIMARY KEY (f_order_id),
    FOREIGN KEY (f_user_id) REFERENCES user(f_user_id)
) COMMENT='注文情報';

-- ==========================
-- 在庫
-- ==========================
CREATE TABLE stock (
    f_stock_id       INT(11) NOT NULL AUTO_INCREMENT COMMENT '在庫ID',
    f_product_id     INT(11) NOT NULL COMMENT '商品ID',
    f_stock_quantity INT(11) NOT NULL DEFAULT 0 COMMENT '在庫数',
    f_update_time    DATETIME NOT NULL COMMENT '更新日時',
    PRIMARY KEY (f_stock_id),
    FOREIGN KEY (f_product_id) REFERENCES product(f_product_id)
) COMMENT='在庫';

-- ==========================
-- 配送
-- ==========================
CREATE TABLE delivery (
    f_delivery_id           INT(11) NOT NULL AUTO_INCREMENT COMMENT '配送ID',
    f_order_id              INT(11) NOT NULL COMMENT '注文ID',
    f_order_info_status     VARCHAR(50) NOT NULL COMMENT '注文情報ステータス',
    f_delivery_date         DATE NULL COMMENT '配送日',
    f_arrival_complete_date DATE NULL COMMENT '到着完了日',
    PRIMARY KEY (f_delivery_id),
    FOREIGN KEY (f_order_id) REFERENCES order_info(f_order_id)
) COMMENT='配送';

-- ==========================
-- レビュー
-- ==========================
CREATE TABLE review (
    f_review_id  INT(11) NOT NULL AUTO_INCREMENT COMMENT 'レビューID',
    f_user_id    INT(11) NOT NULL COMMENT 'ユーザID',
    f_product_id INT(11) NOT NULL COMMENT '商品ID',
    f_evaluation INT(1) NOT NULL COMMENT '評価（1〜5）',
    f_comment    TEXT NULL COMMENT 'コメント',
    f_post_time  DATETIME NOT NULL COMMENT '投稿日時',
    PRIMARY KEY (f_review_id),
    FOREIGN KEY (f_user_id) REFERENCES user(f_user_id),
    FOREIGN KEY (f_product_id) REFERENCES product(f_product_id)
) COMMENT='レビュー';

-- ==========================
-- カート
-- ==========================
CREATE TABLE cart (
    f_cart_id     INT(11) NOT NULL AUTO_INCREMENT COMMENT 'カートID',
    f_user_id     INT(11) NOT NULL COMMENT 'ユーザID',
    f_product_id  INT(11) NOT NULL COMMENT '商品ID',
    f_quantity    INT(11) NOT NULL DEFAULT 1 COMMENT '数量',
    f_adding_time DATETIME NOT NULL COMMENT '追加日時',
    PRIMARY KEY (f_cart_id),
    FOREIGN KEY (f_user_id) REFERENCES user(f_user_id),
    FOREIGN KEY (f_product_id) REFERENCES product(f_product_id)
) COMMENT='カート';

-- ==========================
-- 注文履歴
-- ==========================
CREATE TABLE order_history (
    f_order_history_id INT(11) NOT NULL AUTO_INCREMENT COMMENT '注文履歴ID',
    f_order_id         INT(11) NOT NULL COMMENT '注文ID',
    f_user_id          INT(11) NOT NULL COMMENT 'ユーザID',
    f_order_status     VARCHAR(50) NOT NULL COMMENT '注文ステータス',
    PRIMARY KEY (f_order_history_id),
    FOREIGN KEY (f_order_id) REFERENCES order_info(f_order_id),
    FOREIGN KEY (f_user_id) REFERENCES user(f_user_id)
) COMMENT='注文履歴';
